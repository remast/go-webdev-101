Variablen, Slices, Loops


                // Variable
                var frank string = "Frank"
                claire := "Claire"

                // Array (fixed length)
                namesArray := [3]string{frank, claire, "Zoe"}

                // Slice (variable length)
                namesSlice := make([]string, 2)
                namesSlice[0] = frank

                // Loop
                for i, name := range namesSlice {
                  fmt.Println("Hello " + name + "!")
                }

Errors


                      // Error as return value
                      func (c Congressman) bribe(amount float64) error {
                        if c.Name != "Peter Russo" {
                            return errors.New("Not corrupt!")
                        }
                        c.AccountBalance += amount
                        return nil
                      }

                      func main() {
                        c := Congressman{Name: "Jackie Sharp", AccountBalance: -10.0}

                        // Handle error
                        err := c.bribe(5000.0)
                        if err != nil {
                            fmt.Printf("%v is not bribable.", c.Name)
                        }
                      }
  • Go has no special concept for error handling
  • errors are normal return values

Health Check simple


            func healthHandler(w http.ResponseWriter, r *http.Request) {
              fmt.Fprintf(w, "Up and running!")
            }
            
            func main() {
              http.HandleFunc("/health", healthHandler)
              http.ListenAndServe(":8080", nil)
            }

Health Check with JSON


            type CheckResult struct {
              Status string
            }
            
            func healthHandler(w http.ResponseWriter, r *http.Request) {

              json.NewEncoder(w).Encode(c)

Test Handler health_test.go


          func TestHealthHandler(t *testing.T) {
            // 1. Create test request
            req, _ := http.NewRequest("GET", "/health", nil)
          
            // 2. Create recorder (which satisfies http.ResponseWriter)
            recorder := httptest.NewRecorder()
          
            // 3. Invoke handler
            healthHandler(recorder, req)
          
            // 4. Check the response
            if status := recorder.Code; status != http.StatusOK {
              t.Errorf("Wrong status: got %v expected %v", status, http.StatusOK)
            }
          }
